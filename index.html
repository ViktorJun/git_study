Gallery


git --version -- чтоб проверить какая версия гита установлена(показывает в целом установлен ли он)
D: -- чтоб зайти на конкретный диск
cd НАЗВАНИЕ ПАПКИ -- чтоб зайти в конкретную папку на диске
dir -- чтоб посмотреть какие вообще есть файлы в папке
git init -- обращаемся к git чтоб выполнить команду init(создает скрытую папку для дальнейшей работы)
cd .. -- чтоб выйти с папки
git status -- показывает что вообще происходит с гитом сейчас(вводить нужно в конкретной папке)
git add НАЗВАНИЕ_ФАЙЛА -- чтоб добавить на отслеживание файл(при изменении файла нужно его заново добавлять)
git rm --cached НАЗВАНИЕ_ФАЙЛА -- чтоб перестать следить за файлом
cls -- чтоб очистить консоль
git add . -- чтоб добавить абсолютно все файлы в отслеживание(при изменении файла нужно его заново добавлять)
git commit -m "НАЗВАНИЕ КОММИТА" -- чтоб создать коммит


ДЛЯ ТОГО ЧТОБ ГИТ НЕ ОБРАЩАЛ ВНИМАНИЯ НА КАКИЕ-ТО ФАЙЛЫ НУЖНО СОЗДАТЬ ФАЙЛ С НАЗВАНИЕМ ".gitignore" И В НЕМ ПРОПИСАТЬ ВСЕ ФАЙЛЫ, КОТОРЫЕ НУЖНО ЗАМУТИТЬ(каждый файл с новой строки)
(А ТАКЖЕ НУЖНО ДОБАВИТЬ ЭТОТ ФАЙЛ ".gitignore" В ОТСЛЕЖИВАНИЕ git add НАЗВАНИЕ_ФАЙЛА)(ЧТОБ ИГНОРИРОВАЛ ПАПКУ, НУЖНО ВВЕСТИ В ".gitignore" /НАЗВАНИЕ_ПАПКИ)

git branch -- чтоб посмотреть ветку
git branch НАЗВАНИЕ_ВЕТКИ -- чтоб создать новую ветку(нужна для отдельного программиста, чтоб тот создавал что-то на своей стороне)(создавая ветку ты сразу не переключаешся на неё)
git branch -D test -- чтоб удалить ветку
git checkout НАЗВАНИЕ_ВЕТКИ -- чтоб переключиться на определенную ветку
git checkout -b НАЗВАНИЕ_ВЕТКИ -- чтоб создать ветку и сразу же на неё переключиться
git merge НАЗВАНИЕ_ВЕТКИ -- чтоб совмещать две ветки(совместиться ветка на которой ты сейчас находишься и указанная ветка)(все файлы просто синхронизируються)
git config --global user.name "ViktorJun" -- чтоб назначить новое имя в гите(если не указывать новое имя, то оно просто выведет старое)
git config --global user.email "viktorbylash@gmail.com" -- чтоб назначить новую почту в гите(если не указывать новую почту, то выведеться старая)

git remote add origin https://github.com/ViktorJun/git_study.git -- строка кода, которую дал гитхаб, после создания репозитория(связывает локальный гит с гитхабом)
git push -u origin main -- заливает всё что есть в локальном гите на гитхаб
ЧТОБ ПРОСТО ЗАЛИВАТЬ ВСЁ НА ГИТХАБ МОЖНО ПИСАТЬ git push(ЕСЛИ ЗАЛИВАЕШЬ НЕ ПЕРВЫЙ РАЗ)

mkdir НАЗВАНИЕ_ПАПКИ -- чтоб создать папку из консоли
git clone ССЫЛКА_ВЗЯТАЯ_С_РЕПОЗИТОРИЯ_НА_ГИТХАБ -- можно клонировать все файлы, залитые на гатхаб на свой компьютер
git pull -- забирает все последние изменения, которые были произведены

git commit --reuse-message=HEAD --Это создаст новую фиксацию с тем же сообщением, что и предыдущая фиксация. Аргумент HEADуказывает Git использовать
самую последнюю фиксацию в качестве источника сообщения фиксации.

git remote set-url origin https://github.com/ViktorJun/git_study.git -- при обонвлении ссылки на репозиторий
